CREATE DATABASE  IF NOT EXISTS `tiiiino` 

USE `tiiiino`;

DROP TABLE IF EXISTS `userinfo`;

CREATE TABLE `userinfo` (
  `UID` int NOT NULL AUTO_INCREMENT COMMENT 'Unique ID',
  `ID` varchar(20) NOT NULL COMMENT 'account ID',
  `nick` varchar(20) NOT NULL DEFAULT 'default' COMMENT 'user nickname',
  `grade` decimal(2,1) NOT NULL DEFAULT '0.0' COMMENT 'mmr',
  `point` int NOT NULL DEFAULT '0',
  `state` int NOT NULL DEFAULT '0',
  PRIMARY KEY (`UID`),
  UNIQUE KEY `ID_UNIQUE` (`ID`)
) ENGINE=InnoDB AUTO_INCREMENT=6 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
